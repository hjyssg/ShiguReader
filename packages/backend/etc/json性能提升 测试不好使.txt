const { Readable } = require('stream');
module.exports.sendJsonByStream = function(res, obj){
    const jsonStream = new Readable({
        read() {
          this.push(JSON.stringify(obj));
          this.push(null);
        },
      });
    
      res.writeHead(200, { 'Content-Type': 'application/json' });
      jsonStream.pipe(res);
}
